pipeline {
    agent any
    
    stages {
        stage('Build') {
            steps {
                echo 'sh 'mvn clean''
            }
        }
        
        stage('Unit and Integration Tests') {
            steps {
                echo 'sh 'mvn test''
                
            }
        }
        
        stage('Code Analysis') {
            steps {
               echo 'sh 'mvn checkstyle:check''
            }
        }
        
        stage('Security Scan') {
            steps {
                echo 'sh 'snyk test''
            }
        }
        
        stage('Deploy to Staging') {
            steps {
                echo 'deploying'
            }
        }
        
        stage('Integration Tests on Staging') {
            steps {
               echo 'sh 'mvn integration-test''
            }
        }
        
        stage('Deploy to Production') {
            steps {
                echo 'mvn production'
            }
        }
    }
}
